// --- 舵机逻辑 ---
void rotateServoSequence() {
  if (isOpening) return;
  isOpening = true; servoState = GO_TO_90; servoTimer = millis();
  myServo.attach(SERVO_PIN);
  myServo.write(MAX_ANGLE);
  Serial.println("【门锁】开门"); 
  if(Blinker.connected()) Blinker.print("门", "开");
}

void handleServoNonBlock() {
  if (!isOpening) return;
  unsigned long now = millis();
  if(servoState==GO_TO_90 && now-servoTimer>=VIBRATE_DELAY){
    if(Blinker.connected()) Blinker.vibrate(); 
    servoState=STAY_90; servoTimer=now;
  }
  else if(servoState==STAY_90 && now-servoTimer>=OPEN_KEEP_TIME){
    myServo.write(MIN_ANGLE); Serial.println("【门锁】关门"); 
    servoState=GO_TO_0; servoTimer=now;
  }
  else if(servoState==GO_TO_0 && now-servoTimer>=MOVE_TIME){
    isOpening=false; servoState=IDLE;
    myServo.detach(); 
  }
}

// --- 灯光逻辑 ---
void handleBreathingLight() {
  if (currentState != STATE_RUNNING) return; 
  if (!lightState) {
    digitalWrite(ONBOARD_LED_PIN, HIGH);
    return;
  }
  static int pwmValue = PWM_MAX;
  static int fadeStep = -15;
  static unsigned long lastFadeTime = 0;
  if (millis() - lastFadeTime >= BREATH_SPEED) {
    lastFadeTime = millis();
    pwmValue += fadeStep;
   if (pwmValue <= PWM_MIN) { pwmValue = PWM_MIN; fadeStep = -fadeStep; } 
   else if (pwmValue >= PWM_MAX) { pwmValue = PWM_MAX; fadeStep = -fadeStep; }
   analogWrite(ONBOARD_LED_PIN, pwmValue);
  }
}

// [修复] 空调状态持久化：保存到 EEPROM
void ac_saveState() {
  EEPROM.begin(EEPROM_Total_SIZE);
  EEPROM.write(EEPROM_AC_ADDR + 0, EEPROM_AC_MAGIC);     // magic byte
  EEPROM.write(EEPROM_AC_ADDR + 1, ac_power ? 1 : 0);
  EEPROM.write(EEPROM_AC_ADDR + 2, ac_mode);
  EEPROM.write(EEPROM_AC_ADDR + 3, ac_temp);
  EEPROM.write(EEPROM_AC_ADDR + 4, (uint8_t)fan_level);
  // EEPROM_AC_ADDR + 5 由季节逻辑自行管理，这里不碰
  EEPROM.commit();
  EEPROM.end();
}

// [修复] 空调状态持久化：从 EEPROM 恢复
bool ac_loadState() {
  EEPROM.begin(EEPROM_Total_SIZE);
  byte magic = EEPROM.read(EEPROM_AC_ADDR + 0);
  if (magic != EEPROM_AC_MAGIC) {
    EEPROM.end();
    return false;  // 无有效数据，使用默认值
  }
  ac_mode   = EEPROM.read(EEPROM_AC_ADDR + 2);
  ac_temp   = EEPROM.read(EEPROM_AC_ADDR + 3);
  fan_level = EEPROM.read(EEPROM_AC_ADDR + 4);
  // [修复] 断电恢复后强制电源为关，防止空调自动开机
  // 仅恢复用户偏好设置（模式/温度/风速），不恢复开关状态
  ac_power  = false;
  // 范围校验，防止 EEPROM 数据损坏导致异常值
  if (ac_temp < 16 || ac_temp > 30) ac_temp = DEFAULT_AC_TEMP;
  if (fan_level < 1 || fan_level > 5) fan_level = DEFAULT_FAN_LEVEL;
  applyFanLevel(fan_level);
  EEPROM.end();
  Serial.printf("[空调] 从 EEPROM 恢复偏好 → %s %d°C %s风 (电源:关)\n", 
    getModeText().c_str(), ac_temp, getFanText().c_str());
  return true;
}

// [修复] 带防抖的红外发送：滑块/模式变化后延迟 500ms 再发送，避免高频冗余指令

  acSendPending = true;

void applyFanLevel(int level) {
  if (level < 1) level = 1; if (level > 5) level = 5;
  fan_level = level;
  switch(fan_level) {
    case 1: ac_fan = kCoolixFanFixed; break;
    case 2: ac_fan = kCoolixFanMin;   break;
    case 3: ac_fan = kCoolixFanMed;   break;
    case 4: ac_fan = kCoolixFanMax;   break;
    case 5: ac_fan = kCoolixFanAuto0; break;
  }
}

String getModeText() {
  switch(ac_mode) {
    case kCoolixAuto: return "自动";
    case kCoolixCool: return "制冷";
    case kCoolixDry:  return "除湿";
    case kCoolixHeat: return "制热";
    default: return "送风";
  }
}

String getFanText() {
  switch(fan_level) {
    case 1: return "静音"; case 2: return "低速";
    case 3: return "中速"; case 4: return "高速"; case 5: return "自动"; default: return "未知";
  }
}

void sendAcState() {
  // [修复] 用户手动修改温度/模式后，锁定当月季节逻辑
  if (_seasonManualEdit) {
    _seasonManualEdit = false;
    EEPROM.begin(EEPROM_Total_SIZE);
    EEPROM.write(EEPROM_AC_ADDR + 5, 255); // 当月锁定，下月才重新建议
    EEPROM.commit();
    EEPROM.end();
    Serial.println("[季节] 用户手动编辑，当月锁定季节建议");
  }

  if (!ac_power) { 
    ac.off(); ac.send(); 
    ac_saveState();  // [修复] 每次发送后持久化
    Serial.println("【空调】已关机"); 
    return; 
  }
  ac.on();
  ac.setMode(ac_mode);
  ac.setTemp(ac_temp);
  ac.setFan(ac_fan);
  ac.send();
  ac_saveState();  // [修复] 每次发送后持久化
  Serial.printf("【空调】已发送 → %s %d°C %s风\n", getModeText().c_str(), ac_temp, getFanText().c_str());
}

void wm_saveConfig(String ssid, String pass) {
  EEPROM.begin(EEPROM_Total_SIZE);
  memset(&wifi_config, 0, sizeof(wifi_config));
  ssid.toCharArray(wifi_config.ssid, 32);
  pass.toCharArray(wifi_config.pass, 64);
  wifi_config.magic = MAGIC_BYTE;
  EEPROM.put(EEPROM_ADDR_START, wifi_config);
  EEPROM.commit();
  EEPROM.end();
}

bool wm_loadConfig() {
  EEPROM.begin(EEPROM_Total_SIZE);
  EEPROM.get(EEPROM_ADDR_START, wifi_config);
  EEPROM.end();
  if (wifi_config.magic == MAGIC_BYTE && String(wifi_config.ssid).length() > 0 && wifi_config.ssid[0] != 255) {
    return true;
  }
  return false;
}

// [新增] WiFi扫描API — 异步返回JSON，避免首页阻塞
// 带缓存：5秒内不重复扫描
String _scanCache = "";
unsigned long _scanCacheTime = 0;
void wm_handleApiScan() {
  server.sendHeader("Access-Control-Allow-Origin", "*");
  
  // 有缓存且未过期（5秒内），直接返回
  if (_scanCache.length() > 0 && millis() - _scanCacheTime < 5000) {
    server.send(200, "application/json; charset=utf-8", _scanCache);
    return;
  }
  
  int n = WiFi.scanNetworks();
  String json = "[";
  if (n > 0) {
    // 索引数组排序
    int cnt = (n > MAX_SCAN_RESULTS) ? MAX_SCAN_RESULTS : n;
    int indices[MAX_SCAN_RESULTS];
    for (int i = 0; i < cnt; i++) indices[i] = i;
    for (int i = 0; i < cnt - 1; i++) {
      for (int j = 0; j < cnt - i - 1; j++) {
        if (WiFi.RSSI(indices[j]) < WiFi.RSSI(indices[j + 1])) {
          int tmp = indices[j]; indices[j] = indices[j + 1]; indices[j + 1] = tmp;
        }
      }
    }
    for (int i = 0; i < cnt; i++) {
      int id = indices[i];
      if (WiFi.SSID(id).length() > 0) {
        if (i > 0) json += ",";
        json += "{\"s\":\"" + WiFi.SSID(id) + "\",\"r\":\"" + String(WiFi.RSSI(id)) + "\"}";
      }
    }
  }
  json += "]";
  
  _scanCache = json;
  _scanCacheTime = millis();
  server.send(200, "application/json; charset=utf-8", json);
}

void wm_connect() {
  currentState = STATE_CONNECTING;
  WiFi.mode(WIFI_STA);
  WiFi.hostname(CUSTOM_HOSTNAME);
  WiFi.persistent(false);
  WiFi.begin(wifi_config.ssid, wifi_config.pass);
  Serial.print("\n[系统] 连接 WiFi: "); Serial.println(wifi_config.ssid);

  unsigned long start = millis();
  unsigned long lastBlink = 0;
  while (millis() - start < CONFIG_CONNECT_TIMEOUT) {
    if (millis() - lastBlink > 100) {
      lastBlink = millis();
      digitalWrite(ONBOARD_LED_PIN, !digitalRead(ONBOARD_LED_PIN));
    }
    if (WiFi.status() == WL_CONNECTED) {
      Serial.println("[系统] WiFi 连接成功!");
      Serial.print("[系统] 局域网控制地址: http://"); Serial.println(WiFi.localIP());
      // [修复] WiFi 连上后不直接调 user_setup()，先让 Blinker.begin() 建立连接
      // Blinker.begin() 内部会重新调 WiFi.begin()，需要等它完成后 WiFi 才真正稳定
      Blinker.begin(AUTH, wifi_config.ssid, wifi_config.pass);
      WiFi.hostname(CUSTOM_HOSTNAME);
      currentState = STATE_BLINKER_INIT;
      Serial.println("[系统] 等待 Blinker 云端连接...");
      return;
    }
    delay(10);
  }
  Serial.println("[系统] 连接超时 -> 转 AP");
  wm_startAP();
}

// [修复 P1] 简化灯光定时逻辑，消除令人困惑的嵌套 static 和冗余条件
void updateLightStateAccordingToTime() {
  unsigned long epoch = timeClient.getEpochTime();
  if (epoch < 1600000000) return;  // NTP 未同步
  int h = timeClient.getHours();
  bool shouldBeOn = (h >= 8 && h < 12) || (h >= 14 && h < 24);

  if (h != lastHour) {
    // 新小时：重置手动覆盖，强制设为目标状态
    lightManualOverride = false;
    lastHour = h;
    lightState = shouldBeOn;
    if(Blinker.connected()) btn_light.print(lightState ? "on" : "off");
  } else if (!lightManualOverride && lightState != shouldBeOn) {
    // 同小时内：仅在无手动覆盖且状态不一致时自动调整
    lightState = shouldBeOn;
    if(Blinker.connected()) btn_light.print(lightState ? "on" : "off");
  }
}

// [季节自动切换] 每月首次开机时根据月份建议空调模式+温度
// 夏(6-9月)制冷26°C | 冬(12/1/2月)制热24°C | 春秋(3-5月/10-11月)自动26°C
// EEPROM_AC_ADDR + 5 存储规则：
//   0     = 从未执行过季节建议（首次使用）
//   1~12  = 该月已给出过季节建议（可能已被用户手动覆盖）
//   255   = 用户已手动编辑，当月锁定，下个月才重新建议
// 流程：开机 → NTP同步 → 检查 +5 字节
//   如果 255 或 当前月 == 存储月 → 跳过（当月已处理）
//   否则 → 给出建议值，存当前月到 +5
// 用户手动调温度/模式时 → 存 255 到 +5（当月锁定）
void autoSetACBySeason() {
  unsigned long epoch = timeClient.getEpochTime();
  if (epoch < 1600000000) {
    Serial.println("[季节] NTP 未同步，跳过季节检查");
    return;
  }

  time_t raw = (time_t)epoch;
  struct tm *t = gmtime(&raw);
  int month = t->tm_mon + 1;

  // 从 EEPROM 读取季节状态标记
  EEPROM.begin(EEPROM_Total_SIZE);
  byte seasonFlag = EEPROM.read(EEPROM_AC_ADDR + 5);
  EEPROM.end();

  // 255 = 用户已手动编辑，当月锁定，跳过
  if (seasonFlag == 255) {
    Serial.printf("[季节] %d月 用户已手动设置，保持当前值 %s %d°C\n", month, getModeText().c_str(), ac_temp);
    return;
  }

  // 当前月已经建议过了，跳过
  if (seasonFlag == (uint8_t)month) {
    Serial.printf("[季节] %d月 已建议过，保持当前值 %s %d°C\n", month, getModeText().c_str(), ac_temp);
    return;
  }

  // 新月份（或首次使用），根据月份确定建议值
  uint8_t new_mode;
  int new_temp;

  if (month >= 6 && month <= 9) {
    new_mode = kCoolixCool;
    new_temp = 26;
  } else if (month == 12 || month == 1 || month == 2) {
    new_mode = kCoolixHeat;
    new_temp = 24;
  } else {
    new_mode = kCoolixAuto;
    new_temp = 26;
  }

  // 更新内存中的建议值（不开机、不发红外，仅更新变量）
  ac_mode = new_mode;
  ac_temp = new_temp;

  // 存储当前月到 EEPROM，标记该月已给出建议
  EEPROM.begin(EEPROM_Total_SIZE);
  EEPROM.write(EEPROM_AC_ADDR + 5, (uint8_t)month);
  EEPROM.commit();
  EEPROM.end();

  const char* sName = (month >= 6 && month <= 9) ? "夏天-制冷26C" :
                      (month == 12 || month == 1 || month == 2) ? "冬天-制热24C" :
                      "春秋-自动26C";
  Serial.printf("[季节] %d月 -> %s (新月份建议值)\n", month, sName);

  if (Blinker.connected()) updateAppState();
}

void user_loop() {
  Blinker.run();
  server.handleClient();

  // [修复] 空调防抖：滑块/模式变化后 500ms 再发送红外
  if (acSendPending && millis() - acSendTimer >= 500) {
    acSendPending = false;
    sendAcState();
  }

  // [修复 P1] static 变量提取到函数顶层，消除嵌套 static 提升可读性
  static unsigned long lastTimeCheck = 0;
  static unsigned long lastForceCheck = 0;
  // [季节] NTP 首次同步后执行一次季节检查（仅一次）
  static bool seasonChecked = false;
  if (millis() - lastTimeCheck > 1000) {
    lastTimeCheck = millis();
    if(timeClient.update()) {
      updateLightStateAccordingToTime();
      // NTP 首次同步成功，执行季节建议检查
      if (!seasonChecked) {
        seasonChecked = true;
        autoSetACBySeason();
      }
    }
    if (millis() - lastForceCheck > 30000) {
      lastForceCheck = millis();
      updateLightStateAccordingToTime();
    }
  }

  if (!mqttClient.connected()) {
    static unsigned long lastMqttRetry = 0;
    if (millis() - lastMqttRetry > 5000) {
      lastMqttRetry = millis();
      String clientId = "ESP8266-" + String(random(0xffff), HEX);
      if (mqttClient.connect(clientId.c_str())) mqttClient.subscribe(mqtt_topic);
    }
  } else {
    mqttClient.loop();
  }
}

    if (millis() - lastBlink > 1000) {
      lastBlink = millis();
      digitalWrite(ONBOARD_LED_PIN, !digitalRead(ONBOARD_LED_PIN));
    }
    dnsServer.processNextRequest();
    server.handleClient();
    
    static unsigned long apStart = 0;
    if (apStart == 0) apStart = millis();
    if (millis() - apStart > CONFIG_AP_TIMEOUT) {
      ESP.restart();
    }
  }
  // --- 1.5 Blinker 云端初始化阶段 ---
  // [修复] WiFi 连上后 Blinker.begin() 内部会重新调 WiFi.begin()
  // 必须持续跑 Blinker.run() 等 WiFi + Blinker 都稳定后才能进入正常运行
  else if (currentState == STATE_BLINKER_INIT) {
    static unsigned long lastBlink = 0;
    static unsigned long initStart = 0;
    if (initStart == 0) initStart = millis();
    if (millis() - lastBlink > 200) {
      lastBlink = millis();
      digitalWrite(ONBOARD_LED_PIN, !digitalRead(ONBOARD_LED_PIN));
    }
    Blinker.run();

    // 等待条件：WiFi 连接稳定 + Blinker 已连接，或者超时 30 秒
    if (Blinker.connected() && WiFi.status() == WL_CONNECTED) {
      currentState = STATE_RUNNING;
      // Blinker 连接成功后初始化业务逻辑
      user_setup();
      Serial.println("[系统] Blinker 就绪，进入运行模式");
    } else if (millis() - initStart > 10000) {
      // [优化] 超时缩至10秒：WiFi有网但无互联网时无需等太久，局域网功能不受影响
      Serial.println("[系统] Blinker 连接超时(10s)，以离线模式运行");
      currentState = STATE_RUNNING;
      user_setup();
    }
  }
  // --- 2. 正常运行模式逻辑 ---
  else if (currentState == STATE_RUNNING) {
    static unsigned long lastCheck = 0;
    
    // [修复] 去掉开机缓冲——进入 STATE_RUNNING 时 Blinker + WiFi 已完全稳定
    
    // 定时巡检
    if (millis() - lastCheck > CONFIG_CHECK_INTERVAL) {
      lastCheck = millis();
      
      if (WiFi.status() != WL_CONNECTED) {
        Serial.println("[监测] 网络波动...");
        delay(500); 
        if (WiFi.status() == WL_CONNECTED) {
           return; 
        }

        Serial.println("[警告] 确认掉线，重连中...");
        int retry = 0;

        // [修复 P1] 非阻塞式重连，保证舵机状态机时序精度
        while (WiFi.status() != WL_CONNECTED && retry < CONFIG_MAX_RETRY) {
          for (int i = 0; i < 100 && WiFi.status() != WL_CONNECTED; i++) {
            delay(10);
            handleServoNonBlock();
          }
          Serial.print("."); retry++;
        }
        
        if (WiFi.status() != WL_CONNECTED) {
          Serial.println("\n[失败] 转 AP 模式");
          wm_startAP();
          return;
        } else {
           Serial.println("\n[恢复] 重连成功!");
           lastCheck = millis(); 
        }
      }
    }
    
    handleBreathingLight(); 
    user_loop();            
  }
  
  delay(5); 
}
